事件傳播模型(event progation)

事件發生時，DOM裡頭任一節點都有機會來處理，處理順序為何？

事件捕捉模型(Event Capturing)

事件發生時由最大的元素開始往下傳遞(document->div->p)

事件氣泡模型(Event Bubbling)

由內往外
大部分情況是用氣泡模型
參數放false為氣泡模型;true則為捕捉模型

<body id = 'bodybody'>
    <div id = 'papa'>
        <button id = 'xd'>bubble</button>
    </div>
    <div id = 'sider'></div>
</body>

//get all element
var btn = document.getElementById('xd');
var papa = document.getElementById('papa');
var sider = document.getElementById('sider');
var bodybody = document.getElementById('bodybody');

//give event of each
//false:bubble;true:capture
btn.addEventListener('click',eventHandler,false);
papa.addEventListener('click',eventHandlerPapa,false);
sider.addEventListener('click',eventHandlerSide,false);
bodybody.addEventListener('click',eventHandlerBody,false);
document.addEventListener('click',eventHandlerDoc,false);

//function
function eventHandler(){
    console.log('hi');
};
function eventHandlerPapa(){console.log('hi papa');};
function eventHandlerSide(){console.log('hi side');};
function eventHandlerBody(){console.log('hi body');};
function eventHandlerDoc(){console.log('hi document');};

動態註冊事件

example1

<ul id='num'></ul>

const list = document.getElementById('num'); 

//dynamic creating by loop
for(var i=0;i<5;i++){
    //dynamic create element
    var option = document.createElement('LI');
    //set attribute into element
    option.setAttribute('class','style');
    option.addEventListener('click',hand,false);

    option.innerHTML = i;
    //create child for "num"
    list.appendChild(option);
}
//dynamic remove element
function hand(){
    var dad = this.parentNode;
    dad.removeChild(this);
}

function eventHandler(){
    var ul = document.getElementById('playlist');
    var songText = document.getElementById('songTextInput');
    var songTextValue = songText.value;

    li.innerHTML = songTextValue;
    ul.appendChild(li);
};
