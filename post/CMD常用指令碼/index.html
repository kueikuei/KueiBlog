純CMD

//產生新資料夾
$mkdir direct_name

//轉移到某路徑下
$cd direct_name

//印出當前路徑
$pwd

//產生某種類型的檔案
$touch file_name

//呼叫出所有路徑下檔案
$ls

//呼叫出所有路徑下檔案(包含隱藏檔案)
$ls -a

//藉由cmd快速寫入資料
$echo '# another hello world!' >> 'file_name'

Hexo CMD

//Create a new post
$hexo new post "My New Post"

//Run hexo server 4000
$hexo server
或
$hexo s

//清除不存在於_posts下的markdown檔案
$hexo clean
或
$hexo c

//依照_posts下的檔案產出其對應html檔案於public/post下
$hexo generate
或
$hexo g

//clean和generate通常一起使用
$hexo c && hexo g 

//部署
$hexo deploy
或
$hexo d

GIT CMD

//控管此路徑下檔案
$git init  

//查看此git使用者資訊
$git config --list 

//加入追蹤對象
$git add  

//commit後加入儲存庫裡頭，並添加commit訊息
$git commit -m"message" 

//連接到遠端儲存庫
$git remote add origin   

//推上遠端儲存庫 
$git push origin master

//記錄每一筆commit，檢視 commit 是否成功
$git log

//圖形流程圖去看
$git log --oneline --graph --all
$git log --graph --all

//查看工作目錄的狀態時...
$git status

//或是看更精簡一點的狀態...
$git status -s

//目前code與上一個版本的差別，還沒放進stage area(還沒git add)
$git diff

//假若已經放進stage area了(已經git add)，則使用此查看差別
git diff --cached

//取消已經暫存的(git add)檔案
$git reset HEAD 'file_name'

//取消對檔案的修改(edit)
$git checkout -- 'file_name'

//恢復 Commit，重新提交 Commit
$git reset --soft HEAD\^

//恢復 Commit，重新提交 Commit -  commit amend
$git add 'file_name'
$git commit --amend

//如果懶得更改commit message，則打入
$git commit --amend --no-edit

//強制恢復到上一版本(commit)
$git reset --hard HEAD\^

//想要回到多個版本之前
$git reset --hard HEAD~'num'

//創立新的分支
$git branch 新分支名稱

//切換到該分支:
$git checkout 新分支名稱

//回到主支
$git checkout master
